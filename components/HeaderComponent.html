import { Router } from '@angular/router';
import { Component, OnInit } from '@angular/core';

import { TiposService } from '../../services/tipos.service';
/**
 * Header component includes the menus.
 */
@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.css']
})

export class HeaderComponent implements OnInit {
  /**
   * Rxjs operation name
   */
  nameRxjs: string = "Rxjs";
  /**
   * Testing operation name
   */
  testing: string = "Testing";
  /**
   * Rxjs operation name
   */
  ngRx: string = "Ngrx";
  
  /**
   * Rxjs options names
   */
  rxjsOptions: string[] = [];
   /**
   * Ngrx options names
   */
  ngrxOptions: string[] = [];
   /**
   * Testing options names
   */
  testingOptions: string[] = [];
  
  /**
   * @ignore
  */  
  types: any;
  /**
   * @ignore
  */  
  componentSel: string;
  /**
   * @ignore
  */  
  elemenSelected: string;

/**
 * The constructor method instantiates the service and the {@link https://angular.io|Router}
 * @param tipoService Service instance to get the types.
 * @param router Instance of the Router for navigation.
 */

  constructor( 
    private tipoService : TiposService,
    private router: Router ){
    this.rxjsOptions= [];
    this.componentSel = "";
    this.elemenSelected = "";
    this.testingOptions.push("Incrementador")

  }
  /**
   * When component starts, service gets the options from the dropdown.
   */
  ngOnInit(): void  {

    this.rxjsOptions = this.tipoService.obtenerTipos();
   
   }

  /**
 * This method capteres the click event and sends the selected element as an argument to the navigate function.
 * {@link http://www.google.com event definition}
 * @example 
 * <div class="col-6" style="text-align: right;">
 *        <app-dropdown [lines]="rxjsOptions" [name]="nameRxjs" (ruta)="captureSelected($event)"></app-dropdown>
 *  <div>
 * @param event event click en el DOM 
 */

  captureSelected(event: string){
    this.elemenSelected = event.toString();
    this.navigateComponent(this.elemenSelected); 
    
  }

  /**
   * {@link https://angular.io/guide/router Navigate}  to others components.
   * @param element name of the selected type.
   */

  navigateComponent(element : string){

    switch (element) {
      case 'Subject':
        this.router.navigate(['/subject']);
        break;

      case 'Forkjoin':
        this.router.navigate(['/forkjoin']);
        break;

      case 'Map':
        this.router.navigate(['/maps']);
        break;

      case 'Observable':
        this.router.navigate(['/observable']);
        break;

      case 'Interval':
        this.router.navigate(['/interval']);
        break;
        
      case 'Filter':
        this.router.navigate(['/filter']);
        break;
      case 'Incrementador':
        this.router.navigate(['/incrementador']);
        break;

      case 'Testing':
        this.router.navigate(['/testing']);
        break;

      default:
        break;
    }

  }

   /**
   * Method used for testing
   */
  getType(){
    this.tipoService.getTypes().subscribe(
      (tips : any) => this.types = tips
    );
  }

}
